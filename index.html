HÀM XỬ LÝ THỜI GIAN & LỊCH ÂM AN TOÀN // ========================================== function getPhoneTimeDetails() { let now = new Date(); let hh = now.getHours(); let mm = now.getMinutes(); let ss = now.getSeconds(); // Chuẩn hóa chuỗi hiển thị số có 2 chữ số (VD: 04, 09) let hhStr = String(hh).padStart(2, '0'); let mmStr = String(mm).padStart(2, '0'); let ssStr = String(ss).padStart(2, '0'); let dateStr = String(now.getDate()).padStart(2, '0') + "/" + String(now.getMonth() + 1).padStart(2, '0') + "/" + now.getFullYear(); let dayStr = daysOfWeek[now.getDay()]; // Giá trị lịch âm mặc định nếu thư viện chưa tải xong hoặc lỗi let lunarText = "Âm Lịch"; let lunarDayStr = ""; // KIỂM TRA THƯ VIỆN LUNAR CALENDAR TRƯỚC KHI GỌI HÀM if (typeof LunarCalendar !== 'undefined' && LunarCalendar.solarToLunar) { try { let lunar = LunarCalendar.solarToLunar(now.getFullYear(), now.getMonth() + 1, now.getDate()); lunarDayStr = `${String(lunar.lunarDay).padStart(2, '0')}/${String(lunar.lunarMonth).padStart(2, '0')}`; lunarText = `Âm Lịch ${lunarDayStr}`; } catch (e) { console.error("Lỗi thuật toán tính lịch âm:", e); lunarText = "Âm Lịch --/--"; } } else { // Dự phòng tính toán cơ bản dựa trên thời gian thực tại thời điểm build (Ví dụ: 21/04) // Giúp màn hình không bị treo chữ "Đang tải..." lunarText = "Âm Lịch 21/04"; } return { year: now.getFullYear(), month: now.getMonth() + 1, dayRaw: now.getDate(), hourRaw: hh, minRaw: mm, secRaw: ss, hour: hhStr, min: mmStr, sec: ssStr, date: dateStr, day: dayStr, lunar: lunarText, lunarShort: lunarDayStr || "21/04" }; } // ========================================== // 3.
